## Setting up Clerk middleware

Once you've created the app, you can follow the NextJS instructions:

<br />

<Image
  src="/clerk/follow_demo.png"
  alt="Screenshot of NextJS Setup instructions for Clerk"
  width={1000}
  height={1000}
/>


This includes: 

1. Install Clerk:

```sh filename="terminal" copy
npm install @clerk/nextjs
```

2. Set up your .env.local variables

Click on the package.json file and then click new file - call this file `.env.local`

3. Set up your middleware

Click on the package.json file and create a new file called `middleware.ts`

### Clerk Auth Wrappers

Next, you can update you main layout.tsx file

```tsx filename="layout.tsx" copy
import type { Metadata } from "next";
import { Inter } from "next/font/google";
import "./globals.css";
import NavBar from "@/components/NavBar";

import { ClerkProvider } from "@clerk/nextjs";

const inter = Inter({ subsets: ["latin"] });

export const metadata: Metadata = {
  title: "NextJS, Clerk, GenAI",
  description: "Alex's First AI Course for LCR Developers",
};

export default function RootLayout({
  children,
}: Readonly<{
  children: React.ReactNode;
}>) {
  return (
    <ClerkProvider>
      <html lang="en" className="h-screen">
        <body className={inter.className + " h-full"}>
          <NavBar />
          {children}
        </body>
      </html>
    </ClerkProvider>
  );
}
```

Also update your NavBar component:

```tsx filename="NavBar.tsx" copy
import Link from "next/link";
import React from "react";
import { Button } from "./ui/button";
import { SignInButton, SignedIn, SignedOut, UserButton } from "@clerk/nextjs";

export default function NavBar() {
  const NavLinks = [
    {
      name: "Home",
      href: "/",
    },
    {
      name: "Contact",
      href: "/contact",
    },
  ];

  return (
    <div className="w-full flex flex-1 justify-end gap-x-4 py-2 pr-4">
      {NavLinks.map((link) => (
        <Link
          key={link.name}
          href={link.href}
          style={{ textDecoration: "none" }}
        >
          <Button>{link.name}</Button>
        </Link>
      ))}
      <SignedOut>
        <div className="bg-slate-800 text-white rounded-md items-center px-4 py-2">
          <SignInButton mode="modal" forceRedirectUrl={"/dashboard"} />
        </div>
      </SignedOut>
      <SignedIn>
        <Link
          href={"/dashboard"}
          style={{ textDecoration: "none" }}
        >
          <Button>Dashboard</Button>
        </Link>
        <UserButton afterSignOutUrl="/" />
      </SignedIn>
    </div>
  );
}
```

Now try it out. Log in to you app and you should be directed to a clerk 'account management' screen.

Click on the name you gave your app at the top right and you'll be redirected back to your app.
